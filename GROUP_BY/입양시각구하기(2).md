```SQL
WITH temporary AS ( SELECT LEVEL - 1 hour
                    FROM dual
                    CONNECT BY LEVEL <= 24
                  )

SELECT T.hour, NVL(A.count, 0) count
FROM temporary T LEFT OUTER JOIN ( SELECT TO_NUMBER(TO_CHAR(datetime, 'HH24')) hour, count(*) count
                                   FROM ANIMAL_OUTS
                                   GROUP BY TO_NUMBER(TO_CHAR(datetime, 'HH24'))
                                 ) A
ON T.HOUR = A.HOUR
ORDER BY T.HOUR
```
